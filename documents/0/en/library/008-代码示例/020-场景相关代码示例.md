# Client Scenarios
#### Pre-loading scene resources
If the relevant preloader is checked in the editor, the content of the checkboxes will be loaded in addition to the basic scene creation data<br>
> AssetManager.preLoadSceneAsset(Scene ID, Callback.New(() => {<br>
&nbsp;&nbsp;&nbsp;&nbsp;// Logic executed after loading is complete<br>
}, this), true, false);<br>

#### Create a scene
It can be used in the main scene, making small maps, multi-scene on the same screen and other places.<br>
==If it is the main scene, you need to record Game.currentScene after entering the new scene.==
>// When listening for the need to go to a new master scene execute:<br>
ClientScene.createScene(Scene ID, Callback.New((scene:ClientScene)=>{<br>
&nbsp;&nbsp;&nbsp;&nbsp;// Record the current scene<br>
&nbsp;&nbsp;&nbsp;&nbsp;Game.currentScene = scene;<br>
&nbsp;&nbsp;&nbsp;&nbsp;// Add to the specified level（Total game level reference [GameLayer](?file=006-API-2D单机内核/002-客户端/026-GameLayer)）<br>
&nbsp;&nbsp;&nbsp;&nbsp;Game.layer.sceneLayer.addChild(scene.displayObject);<br>
&nbsp;&nbsp;&nbsp;&nbsp;// Scene starts rendering (if it does not start rendering then the scene is at rest)<br>
&nbsp;&nbsp;&nbsp;&nbsp;Game.currentScene.startRender();<br>
}, this), null, true);<br>

#### Control the shots on the scene
The camera jumps to the specified scene location (e.g. pixel coordinate point 500,500):<br>
>// First unlock the target, if the target is not locked, you can not call this code<br>
>Game.currentScene.camera.sceneObject = null;<br>
>// Then set the specified location<br>
>Game.currentScene.camera.viewPort.x = 500;<br>
>Game.currentScene.camera.viewPort.y = 500;<br>
>

Lens-bound scene objects
>// The camera locks the scene object with ID=1 in the current scene (can be optimized to jog to his coordinates here first and then lock)<br>
// The binding is real-time following, if you want to create smooth following and other effects can not use the binding lens, but custom control lens coordinates.<br>
>Game.currentScene.camera.sceneObject = Game.currentScene.sceneObjects[1];<br>

#### Scene layer manipulation
Create an image layer to add to the current scene<br>
>Create a layer and set the image, add it to the scene, the default is in the upper left corner (0,0)<br>
>var layer = new ClientSceneLayer(Game.currentScene);<br>
>layer.setBigImage("asset/image/xxxx.png");<br>
>Game.currentScene.addLayer(layer);<br>
>

<br>
Create a block layer and draw the block<br>
>// Create a block layer<br>
>var layer = new ClientSceneLayer(Game.currentScene);<br>
>layer.drawMode = true;<br>
>layer.graphics.drawRect(0,0,100,100,"#FF0000");<br>
>// Load the specified mapping as a block material<br>
>AssetManager.loadImage("asset/image/tile/矿洞.png", Callback.New((tex: Texture) => {<br>
>&nbsp;&nbsp;&nbsp;&nbsp;// Get the 48x48 map from the map source 0,0 and draw it to the position of grid 3,0<br>
>&nbsp;&nbsp;&nbsp;&nbsp;layer.drawTile(3, 0, { tex: tex, texID: 1, x: 0, y: 0, w: 48, h: 48 });<br>
>&nbsp;&nbsp;&nbsp;&nbsp;// Get the 48x48 plot from the plot source 96,0 and draw it to the grid 4,0<br>
>&nbsp;&nbsp;&nbsp;&nbsp;layer.drawTile(4, 0, { tex: tex, texID: 1, x: 96, y: 0, w: 48, h: 48 });<br>
>&nbsp;&nbsp;&nbsp;&nbsp;// Submit to draw<br>
>&nbsp;&nbsp;&nbsp;&nbsp;layer.flushTile();<br>
>}, this));<br>
>// Add to the scene<br>
>Game.currentScene.addLayer(layer);<br>
>

### Get the scene object on the scene
>// Get the scene object with ID=5<br>
var so:ClientSceneObject = Game.currentScene.sceneObjects[5] as ClientSceneObject ;




# Network version server-side scenarios
The scenes on the server side are created due to the engine, and all scenes and the scene objects on the scenes are created during the engine initialization.<br>
The project layer just needs to write the implementation of the scene binding class, such as how to handle when the player enters the scene and how to handle when the player leaves the scene<br>
Reference example: Project layer server-side scene binding class for GameCreator2DNet kernel:<br>
>/\*\*<br>
 \* Project layer server scenarios<br>
 \* Created by 黑暗之神KDS on 2019-08-07 13:23:30.<br>
 \*/<br>
class GameServerScene extends ServerScene {<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Constructors<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;constructor() {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;super();<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;//------------------------------------------------------------------------------------------------------<br>
&nbsp;&nbsp;&nbsp;&nbsp;// Overload methods for synchronization to the client<br>
&nbsp;&nbsp;&nbsp;&nbsp;//------------------------------------------------------------------------------------------------------<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Add Player<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param player Player<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;addPlayer(player: ServerPlayer): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Inherit the function of this method from the parent class<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;super.addPlayer(player);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Send information about other players on the current scene to the player, calling GameGate::inNewScene of the client.<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;this.sendToPlayer(player, "inNewScene", [this.getAroundSceneObjects(player.sceneObject, false)]);<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Remove player<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param player Player<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;removePlayer(player: ServerPlayer): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Inherit the function of this method from the parent class<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;super.removePlayer(player);<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Adding game objects and synchronizing them to the client<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param so Game objects to be added <br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @return [number] The position of the game object in the list of sceneObjects on the scene<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;addSceneObject(so: ServerSceneObject, addToList: boolean = true): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Inherit the function of this method from the parent class<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;super.addSceneObject(so, addToList);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Transferring data: different data is transferred depending on whether it is the player itself or not<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var selfSyncClientData = so.syncClientData(true);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var elseSyncClientData = so.syncClientData(false);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;for (var uid in this.playerMap) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var player: ServerPlayer = this.playerMap[uid];<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var syncClientData = player == so.player ? selfSyncClientData : elseSyncClientData;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;ServerMsgSender.rpc(player, "GameGate", "addSceneObject", [syncClientData]);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Remove the game object and synchronize it to the client<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param so Game objects to be removed<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param removeFromList [Optional] Default Value=true <br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param force [Optional] Default Value=false <br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;removeSceneObject(so: ServerSceneObject, removeFromList: boolean = true, force: boolean = false): boolean {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var isRemode = super.removeSceneObject(so, removeFromList, force);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (isRemode) this.radioPlayers("removeSceneObject", [{ index: so.index }]);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;return isRemode;<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;//------------------------------------------------------------------------------------------------------<br>
&nbsp;&nbsp;&nbsp;&nbsp;// New method added to this class: Client request<br>
&nbsp;&nbsp;&nbsp;&nbsp;//------------------------------------------------------------------------------------------------------<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Client request to enter the scenario<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param player Requested players<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;protected static requestInScene(player: ServerPlayer): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Since there is currently only one project level scene class, it is 100% certain that the player is that class - GameServerScene<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var scene: GameServerScene = ServerScene.getScene(player.sceneID) as any;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Ignore if no scene exists<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (!scene) return;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Add Player<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;scene.addPlayer(player);<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Client requests for character movement<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param player Requested players <br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param toX Mobile Destinations<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param toY Mobile Destinations<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;protected static requestMove(player: ServerPlayer, toX: number, toY: number): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Since there is currently only one project level scene class, it is 100% certain that the player is that class - GameServerScene<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var scene: GameServerScene = ServerScene.getScene(player.sceneID) as any;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (!scene) return;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// Forwarding messages to all players in the current scene: Call the GameGate::rpcSceneObjectMove method directly from all player clients.<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;scene.radioPlayers("rpcSceneObjectMove", [player.sceneObject.index, toX, toY]);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;// The server records the status of the player's location so that if a player comes online in the middle of the day, the player's location is known.<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;player.sceneObject.x = toX;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;player.sceneObject.y = toY;<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;//------------------------------------------------------------------------------------------------------<br>
&nbsp;&nbsp;&nbsp;&nbsp;// New methods added to this class: Communication and other<br>
&nbsp;&nbsp;&nbsp;&nbsp;//------------------------------------------------------------------------------------------------------<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Broadcast to all players<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param funcName Method name<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param params Parameters<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;radioPlayers(funcName: string, params: any[], rpcClass: string = "GameGate"): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;for (var uid in this.playerMap) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var player: ServerPlayer = this.playerMap[uid];<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;ServerMsgSender.rpc(player, rpcClass, funcName, params);<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Send to the specified player<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param player Designated players<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param funcName Method name<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param params Parameters<br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;sendToPlayer(player: ServerPlayer, funcName: string, params: any[], rpcClass: string = "GameGate"): void {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;ServerMsgSender.rpc(player, rpcClass, funcName, params);<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;/\*\*<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* Get the surrounding scene objects: here as an example directly get all the player objects (if it is an MMORPG then it should be optimized, for example using the AOI algorithm)<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param targetSceneObject Target scene object<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @param includeSelf [Optional] Default Value=false Whether to include yourself (i.e., the target object)<br>
&nbsp;&nbsp;&nbsp;&nbsp; \* @return [SceneObject] <br>
&nbsp;&nbsp;&nbsp;&nbsp; \*/<br>
&nbsp;&nbsp;&nbsp;&nbsp;protected getAroundSceneObjects(targetSceneObject: SceneObject, includeSelf: boolean): SceneObject[] {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var len = this.sceneObjects.length;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var arr = [];<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;for (var i = 0; i < len; i++) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;var so = this.sceneObjects[i] as ServerSceneObject;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (so) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (so == targetSceneObject && !includeSelf) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;continue;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if (so.inScene) arr.push(so.syncClientData(false));<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;return arr;<br>
&nbsp;&nbsp;&nbsp;&nbsp;}<br>
}<br>
